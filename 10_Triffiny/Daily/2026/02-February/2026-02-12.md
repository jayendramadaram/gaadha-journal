// --- CONFIGURATION: SCORING RULES ---
const page = dv.current();
const tasks = page.file.tasks;

// Rule 1: Physiology (Must complete 'Physiology' to get points)
const physio = tasks.filter(t => t.text.includes("Physiology") && t.completed).length > 0 ? 10 : 0;

// Rule 2: Strategy (Must complete Chess)
const strat = tasks.filter(t => t.text.includes("Strategy") && t.completed).length > 0 ? 10 : 0;

// Rule 3: Intelligence (If ANY sub-task like Twitter/Scholar is done, get 10 pts)
// We look for tasks under the Intelligence header section (approximate logic)
const intelTasks = tasks.filter(t => (t.text.includes("Twitter") || t.text.includes("Google scholar") || t.text.includes("Medium") || t.text.includes("emails")) && t.completed);
const intel = intelTasks.length >= 2 ? 10 : (intelTasks.length >= 1 ? 5 : 0); // 10pts for 2+, 5pts for 1

// Rule 4: Skill (Any 1 technical task = 10 pts)
const skillTasks = tasks.filter(t => (t.text.includes("DSA") || t.text.includes("Leetcode") || t.text.includes("Cryptography") || t.text.includes("Distributed")) && t.completed);
const skill = skillTasks.length > 0 ? 10 : 0;

// Rule 5: Build & Record (Standard)
const build = tasks.filter(t => t.text.includes("Build:") && t.completed).length > 0 ? 10 : 0;
const record = tasks.filter(t => t.text.includes("Record:") && t.completed).length > 0 ? 10 : 0;

// CALCULATION
const auto_score = physio + strat + intel + skill + build + record;
const effort = page.effort_score || 0; 
const total = auto_score + effort; // Max 100

// RENDER
const color = total >= 80 ? "#4caf50" : (total >= 50 ? "#ff9800" : "#f44336");
dv.paragraph(`**OPERATIONAL CAPACITY: ${total}%** (System: ${auto_score} | Effort: ${effort})`);
dv.paragraph(`<progress value="${total}" max="100" style="width: 100%; height: 20px; accent-color: ${color};"></progress>`);

// TRAILING HISTORY
const history = dv.pages('"10_Journal/Daily"')
    .where(p => p.file.name != dv.current().file.name)
    .sort(p => p.file.name, 'desc')
    .limit(3)
    .map(p => {
        const s = (p.checklist_score || 0) + (p.effort_score || 0);
        return `| [[${p.file.name}|${p.file.name.split("-").slice(1).join("/")}]] | **${s}%** |`;
    });

if (history.length > 0) {
    dv.paragraph(`\n| Date | Score |\n|---|---|\n${history.join("\n")}`);
}
